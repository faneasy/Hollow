staticUrl

生成静态url

{{ 'style/hollow.css' | staticUrl }}

#

swig = require 'swig'
config = require '../../config'

swig.setFilter 'staticUrl', (input) ->
    url = input.toString()
    if '' != config.staticHost
        return "//#{config.staticHost}/#{url}"
    else
        return url

module.exports = swig
