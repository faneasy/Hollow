 * ------------------
 *
 * ## staticUrl
 * 生成静态url
 * ```html
 * {{ 'style/hollow.css' | staticUrl }}
 * ```
 */

(function() {
  var config, swig;

  swig = require('swig');

  config = require('../config');

  swig.setFilter('staticUrl', function(input) {
    var url;
    url = input.toString();
    if ('' !== config.staticHost) {
      return "//" + config.staticHost + "/" + url;
    } else {
      return url;
    }
  });

  module.exports = swig;

}).call(this);
